    public boolean existeUsuarioRegistrado(String dni) throws Exception {
        String sql = "SELECT id_usuario FROM usuario WHERE dni = ?";
        PreparedStatement pst = con.prepareStatement(sql);
        pst.setString(1, dni);
        ResultSet rs = pst.executeQuery();
        return rs.next();
    }
